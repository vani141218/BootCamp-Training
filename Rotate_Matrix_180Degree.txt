
#include <stdio.h>
int main() 
{
    int r,c;
    printf("Enter the r and c:");
    scanf("%d %d", &r,&c);
    int arr[r][c];
    printf("Enter elements:");
    for(int i=0;i<r;i++)
    {
        for(int j=0;j<c;j++)
        {
            scanf("%d",&arr[i][j]);
        }
    }
    // Step 1 Column Swap for upside down of matrix.
    for(int i=0;i<c;i++)
    {
        int top = 0;
        int bottom = r-1;
        while(top < bottom)
        {
            int temp = arr[top][i];
            arr[top][i] = arr[bottom][i];
            arr[bottom][i] = temp;
            top++;
            bottom--;
        }
    }
    // Step 2 Row Swap from front to back.
    for(int i=0;i<r;i++)
    {
        int top = 0;
        int bottom = c-1;
        while(top < bottom)
        {
            int temp = arr[i][top];
            arr[i][top] = arr[i][bottom];
            arr[i][bottom] = temp;
            top++;
            bottom--;
        }
    }
    for(int i=0;i<r;i++)
    {
        for(int j=0;j<c;j++)
        {
            printf("%d",arr[i][j]);
        }
        printf("\n");
    }
    
    
}
