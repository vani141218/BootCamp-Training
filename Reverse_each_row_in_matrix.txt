// Reverse each row in a matrix.
#include <stdio.h>
int main() 
{
    int r,c;
    printf("Enter the r and c:");
    scanf("%d %d", &r,&c);
    int arr[r][c];
    printf("Enter elements:");
    for(int i=0;i<r;i++)
    {
        for(int j=0;j<c;j++)
        {
            scanf("%d",&arr[i][j]);
        }
    }
    int top = 0,bottom = c-1;
    for(int i=0;i<r;i++)
    {
        while(top < bottom)
        {
            int temp = arr[i][top];
            arr[i][top] = arr[i][bottom];
            arr[i][bottom] = temp;
            top++;
            bottom--;
        }
    }
    
    for(int i=0;i<r;i++)
    {
        for(int j=0;j<c;j++)
        {
            printf("%d",arr[i][j]);
        }
        printf("\n");
    }
}
