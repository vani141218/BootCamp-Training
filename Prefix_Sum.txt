#include<stdio.h>
int main()
{
    int n;
    printf("Enter n:");
    scanf("%d",&n);
    int arr[n];
    printf("Enter elements:");
    for(int i=0;i<n;i++)
    {
        scanf("%d",&arr[i]);
    }
    printf("Enter l and r:");
    int l,r;
    scanf("%d%d",&l,&r);
    int prefix[n];
    prefix[0] = 0;
    for(int i=0;i<n;i++)
    {
        prefix[i] = prefix[i-1] + arr[i-1];
    }
    
    for(int i=0;i<n;i++)
    {
        printf("%d",prefix[i]);
    }printf(",");
    printf("Result is: %d",prefix[r] - prefix[l-1]);
}
