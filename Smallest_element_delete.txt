#include <stdio.h>

int main() {
    int n;
    printf("Enter n value:");
    scanf("%d",&n);
    int arr[n];
    for(int i=0;i<n;i++)
    {
        scanf("%d",&arr[i]);
    }
    int min = arr[0];
    for(int i=0;i<n;i++)
    {
        if(arr[i] < min)
        {
            min = arr[i];
        }
    }
    printf("Deketed element is: %d",min);
}
