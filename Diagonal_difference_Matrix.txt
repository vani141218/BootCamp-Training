
#include <stdio.h>
int main() 
{
    int r,c;
    int n=c;
    printf("Enter the r and c:");
    scanf("%d %d", &r,&c);
    int arr[r][c];
    printf("Enter elements:");
    for(int i=0;i<r;i++)
    {
        for(int j=0;j<c;j++)
        {
            scanf("%d",&arr[i][j]);
        }
    }
    int sum1 =0,sum2=0;
    for(int i=0;i<r;i++)
    {
        for(int j=0;j<c;j++)
        {
            if(i == j)
            {
                sum1+= arr[i][j];
            }
        }
    }
    for(int i=0;i<r;i++)
    {
        for(int j=0;j<c;j++)
        {
            if(i + j == r-1)
            {
                sum2+= arr[i][j];
            }
        }
    }
    printf("Differenece is:%d",sum2 - sum1);
}
