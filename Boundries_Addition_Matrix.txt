#include <stdio.h>
int main() 
{
    int r,c;
    printf("Enter the r and c:");
    scanf("%d %d", &r,&c);
    int arr[r][c],sum1=0,sum2=0,sum3=0,sum4=0;
    printf("Enter elements:");
    for(int i=0;i<r;i++)
    {
        for(int j=0;j<c;j++)
        {
            scanf("%d",&arr[i][j]);
        }
    }
    for(int i=0;i<r;i++)
    {
        for(int j=0;j<c;j++)
        {
            if(i==0)
            {
                sum1+=arr[i][j];
            }
        }
    }
    for(int i=0;i<r;i++)
    {
        for(int j=0;j<c;j++)
        {
            if(j==0)
            {
                sum2+=arr[i][j];
            }
        }
    }
    for(int i=0;i<r;i++)
    {
        for(int j=0;j<c;j++)
        {
            if(j== r-1)
            {
                sum3+=arr[i][j];
            }
        }
    }
    for(int i=0;i<r;i++)
    {
        for(int j=0;j<c;j++)
        {
            if(i== r-1)
            {
                sum4+=arr[i][j];
            }
        }
    }
    
    printf("%d",sum1+sum2+sum3+sum4);
}
