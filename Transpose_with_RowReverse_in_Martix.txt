// Transpose with Row Reverse in Martix.
#include <stdio.h>
int main() 
{
    int r,c;
    printf("Enter the r and c:");
    scanf("%d %d", &r,&c);
    int arr[r][c];
    printf("Enter elements:");
    for(int i=0;i<r;i++)
    {
        for(int j=0;j<c;j++)
        {
            scanf("%d",&arr[i][j]);
        }
    }
    // Transpose matrix..
    int trans[c][r];
    for(int i=0;i<r;i++)
    {
        for(int j=0;j<c;j++)
        {
            trans[j][i] = arr[i][j];
        }
    }
    
    
    for(int i=0;i<r;i++)
    {
        int top =0;
        bottom = c-1;
        while(top < bottom)
        {
            int temp = trans[i][top];
            trans[i][top] = trans[i][bottom];
            trans[i][bottom] = temp;
            top++;
            bottom--;
        }
    }
    
    for(int i=0;i<r;i++)
    {
        for(int j=0;j<c;j++)
        {
            printf("%d",trans[i][j]);
        }
        printf("\n");
    }
    
    
}
